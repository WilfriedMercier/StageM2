\section{IFS \& MUSE}
\begin{frame}{Integral Field Spectroscopy \& MUSE}

	\begin{columns}[T]
	
		\begin{column}{0.7\linewidth}
			\underline{IFS:}
			\begin{itemize}[label=$\rhd$]
				\item 3D cubes (2D spatial + 1D spectral)
				\begin{itemize}[label=$\rightarrow$]
					\item \textbf{photometry + kinematics}
				\end{itemize}
			\end{itemize}					
		
			\vspace{15pt}
			\underline{MUSE:}
			\begin{itemize}[label=$\rhd$]
				\item $1 \times \SI{1}{arcmin\squared}$ FoV $\rightarrow$ \textbf{environment}
				\item $\SI{0.2}{arcsec}$ spatial sampling
				\item spectral range $[\SI{4650}{\angstrom}, \SI{9300}{\angstrom}]$
				\item seeing-limited or AO observations
				\item high sensitivity
				\begin{itemize}[label=$\rightarrow$]
					\item \textbf{low-mass galaxies + blind surveys}
				\end{itemize}								
				
			\end{itemize}
		\end{column}

		\begin{column}{0.4\linewidth}
			\begin{figure}
				\vspace{-15pt}
				\centering
				\includegraphics[width=\linewidth]{{graphics/MUSE_tcontini2}.jpg}
				\caption{MUSE instrument. Credit: Contini Thierry (IRAP)}
			\end{figure}
		\end{column}
	\end{columns}

\end{frame}
